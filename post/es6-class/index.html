es6 class
📅 2021-03-25 👦 Tomtom Young 📖 1289字 ⏱ 3分钟

ES6提供了更接近传统语言的写法，引入了Class类的概念，作为对象的模板，通过class关键字，可以定义类。

基本上，ES6的class可以看作是一个语法糖，他的绝大部分功能ES5都可以做到，只是新的class写法让对象原型的写法更加清晰，更像面向对象编程的语法而已。

类的基本使用

 1// ES5
 2// 手机
 3function Phone(brand, price){
 4    this.brand = brand;
 5    this.price = price;
 6}
 7
 8// 添加方法
 9Phone.prototype.call = function(){
10    console.log("我可以打电话!!");
11}
12
13// 实例化对象
14let Huawei = new Phone('华为', 5999);
15Huawei.call();
16console.log(Huawei);
17
18
19// ES6
20// class
21class Shouji{
22    //构造方法 名字不能修改
23    constructor(brand, price){
24        this.brand = brand;
25        this.price = price;
26    }
27    //方法必须使用该语法, 不能使用 ES5 的对象完整形式
28    call(){
29        console.log("我可以打电话!!");
30    }
31}
32
33let onePlus = new Shouji("1+", 1999);
34
35console.log(onePlus);

类的继承

ES5

 1//手机
 2function Phone(brand, price){
 3    this.brand = brand;
 4    this.price = price;
 5}
 6
 7Phone.prototype.call = function(){
 8    console.log("我可以打电话");
 9}
10
11//智能手机
12function SmartPhone(brand, price, color, size){
13    Phone.call(this, brand, price);
14    this.color = color;
15    this.size = size;
16}
17
18//设置子级构造函数的原型
19SmartPhone.prototype = new Phone;
20SmartPhone.prototype.constructor = SmartPhone;
21
22//声明子类的方法
23SmartPhone.prototype.photo = function(){
24    console.log("我可以拍照")
25}
26
27SmartPhone.prototype.playGame = function(){
28    console.log("我可以玩游戏");
29}
30
31const chuizi = new SmartPhone('锤子',2499,'黑色','5.5inch');
32
33console.log(chuizi);

ES6

 1class Phone{
 2    //构造方法
 3    constructor(brand, price){
 4        this.brand = brand;
 5        this.price = price;
 6    }
 7    //父类的成员属性
 8    call(){
 9        console.log("我可以打电话!!");
10    }
11}
12
13class SmartPhone extends Phone {
14    //构造方法
15    constructor(brand, price, color, size){
16        super(brand, price);// Phone.call(this, brand, price)
17        this.color = color;
18        this.size = size;
19    }
20
21    photo(){
22        console.log("拍照");
23    }
24
25    playGame(){
26        console.log("玩游戏");
27    }
28	// 子类对父类进行重写
29    call(){
30        console.log('我可以进行视频通话');
31    }
32}
33
34const xiaomi = new SmartPhone('小米',799,'黑色','4.7inch');
35// console.log(xiaomi);
36xiaomi.call();
37xiaomi.photo();
38xiaomi.playGame();

类的get和set

 1// get 和 set  
 2class Phone{
 3    get price(){
 4        console.log("价格属性被读取了");
 5        return 'iloveyou';
 6    }
 7
 8    set price(newVal){
 9        console.log('价格属性被修改了');
10    }
11}
12
13//实例化对象
14let s = new Phone();
15
16//获取s.price时会调用get price(){}这个函数，而且这个函数的返回值就是s.price的值
17console.log(s.price);
18//给s.price赋值的时候会调用set price(newVal){}这个函数
19s.price = 'free';
