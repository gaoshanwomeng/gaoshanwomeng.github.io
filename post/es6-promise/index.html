promise是ES6引入的异步编程的新解决方案，语法上promise是一个构造函数，用来封装异步操作并可以获取其成功或者失败的结果。

  1. Promise构造函数：Promise(excutor){ }
  2. Promise.prototype.then 方法
  3. Promise.prototype.catch 方法

实例化Promise对象

 1//实例化 Promise 对象
 2const p = new Promise(function(resolve, reject){
 3    setTimeout(function(){
 4        // let data = '数据库中的用户数据';
 5        // resolve
 6        // resolve(data);
 7
 8        let err = '数据读取失败';
 9        reject(err);
10    }, 1000);
11});
12
13//调用 promise 对象的 then 方法
14p.then(function(value){
15    console.log(value);
16}, function(reason){
17    console.error(reason);
18})

Promise封装AJAX

 1// 接口地址: https://api.apiopen.top/getJoke
 2const p = new Promise((resolve, reject) => {
 3    //1. 创建对象
 4    const xhr = new XMLHttpRequest();
 5    //2. 初始化
 6    xhr.open("GET", "https://api.apiopen.top/getJ");
 7    //3. 发送
 8    xhr.send();
 9    //4. 绑定事件, 处理响应结果
10    xhr.onreadystatechange = function () {
11        //判断
12        if (xhr.readyState === 4) {
13            //判断响应状态码 200-299
14            if (xhr.status >= 200 && xhr.status < 300) {
15                //表示成功
16                resolve(xhr.response);
17            } else {
18                //如果失败
19                reject(xhr.status);
20            }
21        }
22    }
23})
24
25//指定回调
26p.then(function(value){
27    console.log(value);
28}, function(reason){
29    console.error(reason);
30});

Promise链式操作

调用 then 方法 then方法的返回结果是 Promise 对象, 对象状态由回调函数的执行结果决定

  1. 如果回调函数中返回的结果是 非 promise 类型的属性, then方法的返回的 Promise 对象状态为成功, 返回值为这个非 promise 类型的属性
  2. 如果回调函数中返回的结果是 promise 对象， then方法的返回的 Promise 对象和这个promise对象一模一样。
  3. 利用这个特性，就可以进行链式操作
 1//创建 promise 对象
 2const p = new Promise((resolve, reject)=>{
 3    setTimeout(()=>{
 4        resolve('用户数据');
 5        // reject('出错啦');
 6    }, 1000)
 7});
 8
 9const result = p.then(value => {
10    console.log(value);
11    //1. 非 promise 类型的属性
12    // return 'iloveyou';
13    //2. 是 promise 对象
14    // return new Promise((resolve, reject)=>{
15    //     // resolve('ok');
16    //     reject('error');
17    // });
18    //3. 抛出错误
19    // throw new Error('出错啦!');
20    throw '出错啦!';
21}, reason=>{
22    console.warn(reason);
23});
24
25//链式调用
26p.then(value=>{
27
28}).then(value=>{
29
30});

 1//引入 fs 模块
 2const fs = require("fs");

 4// 回调地狱，不仅难以阅读，而且变量容易重名

14//使用 promise 实现
15const p = new Promise((resolve, reject) => {
16    fs.readFile("./resources/为学.md", (err, data) => {
17        resolve(data);
18    });
19});
20
21p.then(value => {
22    return new Promise((resolve, reject) => {
23        fs.readFile("./resources/插秧诗.md", (err, data) => {
24            resolve([value, data]);
25        });
26    });
27}).then(value => {
28    return new Promise((resolve, reject) => {
29        fs.readFile("./resources/观书有感.md", (err, data) => {
30            //压入
31            value.push(data);
32            resolve(value);
33        });
34    })
35}).then(value => {
36    console.log(value.join('\r\n'));
