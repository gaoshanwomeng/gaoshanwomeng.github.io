ES6 引入 rest 参数，用于获取函数的实参，用来代替 arguments。

 1
 2// ES5 获取实参的方式
 3function date(){
 4	console.log(arguments);
 5}
 6date('白芷','阿娇','思慧');
 7
 8// rest 参数
 9function date(...args){
10	console.log(args); // filter some every map 
11}
12date('阿娇','柏芝','思慧');
13
14//rest 参数必须要放到参数最后
15function fn(a,b,...args){
16    console.log(a);
17    console.log(b);
18    console.log(args);
19}
20fn(1,2,3,4,5,6);
